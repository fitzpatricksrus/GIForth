: :	( -- word true | false)
 	// get word name
 	nextToken
 	// start compiling
 	compileBegin
	do
 		nextToken str2Int if
			(PUSH_NEXT_CELL) (PUSH_NEXT_CELL) compileTOS
			compileTOS
		else
			findWord if
				compileTOS
			else
				CompilerWords searchVocab if
					execute
				else
					compileEnd freeWord
					printString
					'?' printChar
					'\n' printChar
					false
					return
				endif
			endif
		endif
	forever
 	;

 : ;
 	compileEnd
  	r> drop
  	true
  	return
  	;

: interp
    do
        nextToken
        str2Int not if
            findWord if
                execute
            else
                printString
                '?' printChar
                '\n' printChar
            endif
        endif
    forever
