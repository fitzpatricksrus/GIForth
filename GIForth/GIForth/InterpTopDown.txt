

: nextToken     // -- stringAddr
: findWord      // stringAddr -- [ wordAddr true | stringAddr false ]
: (jumpIfFalse) // bool --
: execute       // wordAddr --
: (jump)
: parseNumber   // stringAddr -- [ value true | string addr false ]
: printString   // stringAddr --
: (pusHNextCell) // -- value
: printChar     // char --

: interp
    do
        nextToken                   // -- stringAddr
        findWord                    // stringAddr -- [ wordAddr true | stringAddr false ]
        if                          // bool --
            // found the word.  Execute it
            execute                 // wordAddr --
        else
            // may be a number
            parseNumber             // stringAddr -- [ value true | string addr false ]
            if                      //
                // it's a number and it's on the stack.  We're done
            else
                printString         // stringAddr
                63 printChar        // '?'
                13 printChar        // '\n'
            endif
        endif
    false until
;
